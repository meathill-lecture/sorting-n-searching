* ppt中的插入排序代码有问题

  ```javascript
  //原来
  for (let i = 1, len = arr.length; i < len; i++) {
    for (let j = 0, jlen = i; j < jlen; j++) {
      if (arr[i] < arr[j]) {
        arr.splice(i, 0);
        arr.splice(j, 0, arr[i]);
        break;
      }
    }
  }

  //正确
  for (let i = 1, len = arr.length; i < len; i++) {
    for (let j = 0, jlen = i; j < jlen; j++) {
      if (arr[i] < arr[j]) {
        var x = arr.splice(i, 0);   //改动处
        arr.splice(j, 0, x[0]);     //改动处
        break;
      }
    }
  }
  ```

* 二分查找代码也有问题

  ```javascript
  //原来
  function find(num, arr) {
    function binarySearch(left, right) {
      if (left > right) {
        return -1;
      }
      if (right - left <= 1) {
        return num > left ? right : left;
      }
      let mid = right + left >> 1;
      if (num <= arr[mid]) {
        return binarySearch(left, mid);
      } else {
        return binarySearch(mid + 1, right);
      }
    }

    return binarySearch(0, arr.length - 1);
  }

  //正确
  function find(num, arr) {
      function binarySearch(left, right) {
          if(num > arr[right]) {         //新添加
              return right+1;
          }
          if (left > right) {
              return -1;
          }
          if (right - left <= 1) {
              return num > arr[left] ? right : left;           //此处left----->arr[left]
          }
          let mid = right + left >> 1;
          if (num <= arr[mid]) {
              return binarySearch(left, mid);
          } else {
              return binarySearch(mid + 1, right);
          }
      }

      return binarySearch(0, arr.length - 1);
  }
  ```
